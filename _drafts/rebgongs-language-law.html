༄༅།།རྨ་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་བོད་ཀྱི་སྐད་ཡིག་བྱ་བའི་སྲོལ་ཡིག

༡༩༩༣ལོའི་ཟླ༥ཚེས༣༠ཉིན་རྨ་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་སྐབས་བཅུ་བའི་མི་དམངས་འཐུས་མི་ཚོགས་ཆེན་ཚང་འཛོམས་གྲོས་ཚོགས་ཐེངས་གསུམཔའི་སྟེང་གྲོས་འཆམ་བྱུང་ཞིང༡༩༩༣ལོའི་ཟླ༩པའི་ཚེས༡༨ཉིན་མཚོ་སྔོན་ཞིང་ཆེན་སྐབས་བརྒྱད་པའི་མི་དམངས་འཐུས་མི་ཚོགས་ཆེན་རྒྱུན་ལས་ཨུ་ཡོན་ལྷན་ཁང་གི་ཐེངས་ལྔ་བའི་ཚང་འཛོམས་གྲོས་ཚོགས་སྟེང་གཏན་ལ་ཕབ།༡༩༩༤ལོའི་ཟླ༡ཚེས༡ཉིན་ནས་བཟུང་ལག་བསྟར་བྱ་རྒྱུ།

དོན་ཚན་དང་པོ། བོད་ཀྱི་སྐད་ཡིག་སློབ་སྦྱོང་དང༌།བཀོལ་སྤྱོད་གོང་སྤེལ་སོགས་ལ་ཤུགས་སྣོན་དང་ཁག་ཐེག་བྱེད་པའི་ཆེད་དུ《 ཀྲུང་ཧྭ་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་རྩ་ཁྲིམས》དང《རྨ་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་ཀྱི་རང་སྐྱོང་སྲོལ་ཡིག》《 ཀྲུང་ཧྭ་མི་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་མི་རིགས་རང་སྐྱོང་ས་ཁུལ་གྱི་བཅའ་ཁྲིམས》བཅས་གཞིར་བཟུང་ཞིང༌། རང་ཁུལ་དངོས་ཡོད་དང་བསྟུན་ནས་སྲོལ་ཡིག་འདི་གཏན་ལ་ཕབ་པ་ཡིན〉

　　དོན་ཚན་གཉིས་པ། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་ཏང་དངརྒྱལ་ཁབ་ཀྱི་མི་རིགས་སྐད་ཡིག་སྲིད་ཇུས་དང་འབྲེལ་ཡོད་ཁྲིམས་ལུགས།ཁྲིམས་སྲོལ་གྱི་གཏན་ཁེལ་སོགས་ལག་ལེན་མཐར་ཕྱིན་དང་མི་རིགས་སོ་སོའི་སྐད་ཡིག་འདྲ་མཉམ་གྱི་གནས་ལུགས་རྒྱུན་འཁྱོངས། རིགས་སོ་སོས་རང་ཉིད་ཀྱི་སྐད་ཡིག་བཀོལ་སྤྱོད་དང་འཕེལ་རྒྱས་གཏོང་བའི་རང་དབང་ཁག་ཐེག་བྱ་རྒྱུ། མི་རིགས་སྐད་ཡིག་གིས་འདྲ་མཉམ་དང་མཐུན་སྒྲིལ་རོགས་སྐྱོར་ཅན་གྱི་སྤྱི་ཚོགས་རིང་ལུགས་མི་རིགས་འབྲེལ་བ་ཇེ་བརྟན་དང་འཕེལ་རྒྱས་སུ་གཏོང་རྒྱུ་དང༌། སྤྱི་ཚོགས་རིང་ལུགས་དངོས་པོའི་དཔལ་ཡོན་དང་བསམ་པའི་དཔལ་ཡོན་ལ་ཤུགས་སྣོན་པར་ཞབས་འདེགས་ཞུ་རྒྱུ།

　　དོན་ཚན་གསུམ་པ། བོད་ཀྱི་སྐད་ཡིག་ནི་རང་སྐྱོང་ཁུལ་གྱིས་མི་རིགས་རང་སྐྱོང་ས་ཁུལ་གྱི་སྲིད་དབང་ལག་བསྟར་བྱེད་པའི་སྐད་ཡིག་གཙོ་བོ་ཞིག་ཡིན་ཞིང༌། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱི་ལས་གནས་ལག་བསྟར་བྱེད་སྐབས་རྒྱ་ཡིག་དང་བོད་ཡིག་གཉིས་མཉམ་དུ་བཀོལ་རྒྱུ།

　　དོན་ཚན་བཞི་བ། རང་སྐྱོང་ཁུལ་གྱིས་བོད་ཀྱི་སྐད་ཡིག་གི་བྱ་བ་ཁྱབ་བརྡལ་དང་མཐོར་འདེགས་རྒྱུན་འཁྱོངས་དང༌། བོད་ཀྱི་ཕུལ་བྱུང་རིག་གནས་ལོ་རྒྱུས་རྒྱུན་འཛིན་གསར་སྤེལ་དང་སྲུང་སྐྱོབ་བྱེད་རྒྱུ། བོད་ཀྱི་སྐད་ཡིག་འཕེལ་རྒྱས་ལ་ཤུགས་སྣོན་དང་བོད་ཀྱི་སྐད་ཡིག་གིས་རང་སྐྱོང་ཁུལ་གྱི་བཅོས་བསྒྱུར་སྒོ་འབྱེད་དང་དཔའ་འབྱོར་འཛུགས་སྐྲུན་ཐད་ནུས་པ་གལ་ཆེན་ཐོན་དུ་འཇུག་རྒྱུ།

　　དོན་ཚན་ལྔ་པ། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་རིགས་སོ་སོའི་ལས་བཟོ་བར་ཕན་ཚུན་གྱི་སྐད་ཡིག་སློབ་སྦྱོང་དང་སྐུལ་འདེད་བྱེད་པར་སློབ་གསོ་གཏོང་རྒྱུ། བོད་རིགས་ལས་བཟོ་པས་མི་རིགས་རང་ཉིད་ཀྱི་སྐད་ཡིག་སློབ་སྦྱོང་དང་བེད་སྤྱོད་བྱེད་པའི་དུས་མཚུངས་སུ་རྒྱལ་ཡོངས་སྤྱི་བཀོལ་གྱི་རྒྱ་སྐད་དང་རྒྱ་ཡིག་ལ་སློབ་སྦྱོང་བྱེད་པ་དང༌། རྒྱ་རིགས་དང་གྲངས་ཉུང་མི་རིགས་གཞན་པའི་ལས་བཟོ་པ་རྣམས་ལ་བོད་ཡིག་སློབ་པར་སྐུལ་འདེད་བྱེད་རྒྱུ།

　　དོན་ཚན་དྲུག་པ། རང་སྐྱོང་ཁུལ་མི་དམངས་སྲིད་གཞུང་གིས་བོད་ཡིག་བྱ་བའི་ཨུ་ཡོན་ལྷན་ཁང་འཛུགས་རྒྱུ་དང༌། དེས་ཁུལ་ཡོངས་ཀྱི་བོད་ཡིག་བྱ་བ་ལ་དོ་དམ་བྱ་རྒྱུ། རང་སྐྱོང་ཁུལ་བོད་ཡིག་བྱ་བའི་ཨུ་ཡོན་ལྷན་ཁང་གི་འགན་ཁུར་ནི།

ཏང་དང་རྒྱལ་ཁབ་ཀྱི་མི་རིགས་སྐད་ཡིག་སྲིད་ཇུས་དྲིལ་སྒྲོག་དང་ལག་ལེན་མཐར་ཕྱིན་བྱེད་རྒྱུ་དང༌། སྲོལ་ཡིག་འདི་ལག་བསྟར་བྱེད་པར་སྐུལ་འདེད་དང་ཞིབ་བཤེར་བྱེད་པ། བཅའ་ཁྲིམས་དང༌། ཁྲིམས་སྲོལ། སྲིད་ཇུས། སྲོལ་ཡིག་བཅས་་གཞིར་བཟུང་་ནས་བོད་ཡིག་བྱ་བ་ལག་བསྟར་དང་གོང་འཕེལ་གཏོང་བའི་འཆར་གཞི་དང་ཐབས་་ལམ་གཏནའཁེལ་བྱ་རྒྱུ།

བོད་ཡིག་སློབ་སྦྱོང་དང་བཀོལ་སྤྱོད་བྱེད་པར་མཛུབ་སྟོན་བྱ་རྒྱུ།

བོད་ཡིག་ཞིབ་འཇུག་དང་ཁྱབ་བརྡལ། རིག་ཚན་བརྗེ་རེས་དང་ཚད་ལྡན་དུ་གཏོང་བ། ཆེད་ལས་མི་སྣ་གསོ་སྐྱོང་གི་ལས་གཞི་རྩ་འཛུགས་བྱེད་རྒྱུ།

གོང་རིམ་ལས་ཁུངས་དང༌། རིམ་པ་མཉམ་པའི་ལས་ཁུངས་ཀྱི་སྤྱི་དོན་ཡིག་ཆ་དང་ཚོགས་འདུའི་ཚགས་ཡིག འབྲེལ་ཡོད་རྒྱུ་ཆ་ལོ་ཙཱ་བྱེད་རྒྱུ།

རང་སྐྱོང་ཁུལ་རང་གི་རྒྱལ་ཁབ་ཀྱི་ལས་ཁུངས་དང་ཁེ་ལས་ལས་ཁུངས་ཀྱི་ཐམ་ཀ་དང༌། མིང་བྱང༌། ཚོང་རྫས་ཀྱི་མིང་སོགས་ཡིག་སྒྱུར་ལྟ་ཞིབ་བྱེད་རྒྱུ།

བོད་ཀྱི་གནའ་རབས་ཡིག་ཚང་འཚོལ་སྡུད་དང༌།ལེགས་སྒྲིག་གི་འགན་ཁུར་རྒྱུ།

བོད་ཀྱི་སྐད་ཡིག་བྱ་བའི་འབྲེལ་ཡོད་ལས་ཁུངས་བར་གྱི་འབྲེལ་བ་དང་བྱ་བ་ལ་སྣ་ཁྲིད་བྱ་རྒྱུ།

　　དོན་ཚན་བདུན་པ། རང་སྐྱོང་ཁུལ་ཁོངས་གཏོགས་རེབ་གོང་དང༌། གཅན་ཚ རྩེ་ཁོག་བཅས་ཀྱི་མི་དམངས་སྲིད་གཞུང་དུ་བོད་ཀྱི་སྐད་ཡིག་བྱ་བའི་གཞུང་ལས་ཁང་གསར་འཛུགས་བྱ་རྒྱུ།

　　དོན་ཚན་བརྒྱད་པ། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱི་ཆབ་སྲིད་དང༌། དཔལ་འབྱོར ། རིག་གནས། སློབ་གསོ། ཚན་རིག འཕྲོད་བསྟེན། གསར་འགྱུར། བརྙན་འཕྲིན། རྩལ་སྦྱོང་བཅས་ཀྱི་ཁྱབ་ཁོངས་ནང་བོད་ཡིག་བཀོལ་སྤྱོད་བྱེད་པར་ཤུགས་སྣོན་རྒྱག་རྒྱུ།

　　དོན་ཚན་བཅུ་བ། རང་སྐྱོང་ཁུལ་ས་གནས་རང་གི་རྒྱལ་ཁབ་ཀྱི་ལས་ཁུངས་དང་ཁེ་ལས་ལས་ཁུངས་ཀྱི་ཐམ་ཀ་དང༌། མིང་བྱང༌། འཛིན་ཡིག བསྒྲགས་ཡིག ཚོང་རྟགས། ཡིག་ཆའི་ཁ་བྱང༌། ཡིག་ཤུབས། གསལ་བརྡ་སོགས་ལ་རྒྱ་བོད་ཡིག་རིགས་གཉིས་ཀ་སྤྱོད་དགོས། རང་སྐྱོང་ནང་ཁུལ་གྱི་ རྫོང་མཁར་དང༌། གྲོང་བརྡལ་སྡེ་བའི་སྲང་ལམ་གཙོ་བོའི་མིང་དང༌། ལམ་རྟགས།ས་མཚམས་ཀྱི་ རྟགས། སྤྱི་སྤྱོད་སྒྲིག་ཆས།འགྲིམ་འགྲུམ་མཚོན་རྟགས། རླངས་འཁོར་གྱི་སྒོ་བྱང་སོགས་ཡི་གེ་མཚོན་དགོས་པ་ཚང་མ་རྒྱ་བོད་ཡིག་རིགས་གཉིས་ཀ་སྤྱོད་དགོས།

　　དོན་ཚན་བཅུ་བདུན། རང་སྐྱོང་ས་ཁུལ་ནང་ཁུལ་གྱི་བོད་རིགས་སློབ་འབྲིང་དང་སློབ་ཆུང་གིས་བོད་ཡིག་སློབ་ཁྲིད་ལ་ཤུགས་སྣོན་བྱེད་པ་དང་མཉམ་དུ་རྒྱའི་སྐད་ཡིག་གི་སློབ་ཁྲིད་ལའང་ཤུགས་སྣོན་རྒྱག་དགོས། བོད་རིགས་སློབ་མ་ཅུང་མང་བའི་སྤྱིར་བཏང་སློབ་འབྲིང་དང་སློབ་ཆུང་དུ་གནས་ཚུལ་དངོས་གཞིར་བཟུང་སྟེ་བོད་ཀྱི་སྐད་ཡིག་གི་སློབ་ཁྲིད་བཀོད་སྒྲིག་བྱེད་རྒྱུ།

　　དོན་ཚན་བཅོ་བརྒྱད། རང་སྐྱོང་ས་ཁུལ་གྱི་མི་རིགས་དགེ་འོས་སློབ་གྲྭར་བོད་རྒྱ་ཡིག་སྐད་གཉིས་སློབ་ཁྲིད་ལ་ཤུགས་སྣོན་བྱེད་དགོས་ལ། བོད་རྒྱ་སྐད་ཡིག་གཉིས་བཀོལ་སྤྱོད་ཐུབ་པའི་དགེ་རྒན་གསོ་སྐྱོང་བྱེད་དགོས།

　　དོན་ཚན་བཅུ་དགུ རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་བོད་སྐད་ཀྱི་རྒྱང་སྒྲོག་དང་བརྙན་འཕྲིན་ལེགས་སྒྲུབ་བྱེད་པ་དང༌། རིམ་བཞིན་རང་གིས་སྒྲུབ་པའི་ལེ་ཚན་ཁ་སྣོན་བྱེད་དགོས་ལ། ཧུར་བརྩོན་གྱིས《རྨ་ལྷོའི་བོད་ཡིག་ཚགས་པར》གསར་བསྒྲིག་བྱེད་དགོས།རང་སྐྱོང་ཁུལ་ནང་ ཁུལ་གྱི་དཔེ་ཚོང་ཁང་དང་སྦྲག་གཏོང་ཁང་གིས་དཔེ་ཆ་དང་ཚགས་པར།དུས་དེབ་སོགས་ཀྱི་འགྲེམ་སྤེལ་དང་སྐྱེལ་གཏོང་གི་བྱ་བ་ལེགས་པོར་སྒྲུབ་སྟེ།རིམ་གྱིས་བོད་ཡིག་དཔེ་ཆའི་རིགས་ཚན་དང་ཁྱབ་ཁོངས་ཇེ་རྒྱས་སུ་གཏོང་དགོས།

　　དོན་ཚན་ཉི་ཤུ། རང་སྐྱོང་ཁུལ་གྱི་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་ཚན་རྩལ་མི་སྣར་ཡར་སྐུལ་བྱེད་པ་དང༌།རྩོམ་རིག་སྒྱུ་རྩལ་གྱི་ལས་བྱེད་མི་སྣས་བོད་སྐད་ཀྱིས་ཚན་རིག་ཞིབ་འཇུག་དང༌། བརྩམས་ཆོས་དང་དཔྱད་རྩོམ་འབྲི་བ་དང༌། རྩོམ་རིག་སྒྱུ་རྩལ་གསར་བཟོ་བྱེད་པ་དང་འཁྲབ་སྟོན་བྱེད་དགོས།

　　དོན་ཚན་ཉེར་གཅིག རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་བོད་ཀྱི་སྐད་ཡིག་གི་ཚན་རིག་ཞིབ་འཇུག་གི་བྱ་བའི་སྣེ་ཁྲིད་ལ་ཤུགས་སྣོན་གཏོང་བ་དང༌། བོད་ཀྱི་སྐད་ཡིག་ཚན་རིག་ཞིབ་འཇུག་བྱ་བ་དེ་ངེས་པར་དུ་བོད་ཀྱི་སྐད་ཡིག་གི་རྨང་གཞི་དང༌། དངོས་མིང་གསར་བ་དང་བརྡ་ཆད་གསར་བ། ཚན་རྩལ་གྱི་མིང་བཀོལ་སྤྱོད་དང་ཚད་ལྡན་དུ་བསྒྱུར་པར་ཁ་གཏོང་རྒྱུ།

　　དོན་ཚན་ཉེར་གཉིས། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་ཧུར་ཐག་གིས་བོད་ཀྱི་ཁྱད་འཕགས་པའི་རིག་གནས་ཤུལ་བཞག་ལེགས་བསྡུ་དང་སྔོག་འདོན། ལེགས་སྒྲིག་བཅས་བྱེད་དགོས་ལ། བོད་སྐད་ཡིག་གི་གནའ་དཔེ་ལ་བདག་གཉེར་བྱེད་དགོས།

　　དོན་ཚན་ཉེར་གསུམ། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་བོད་ཀྱི་སྐད་ཡིག་བྱ་བའི་དཔུང་སྡེ་འཛུགས་སྐྲུན་ལ་ཤུགས་སྣོན་དང༌། རྣམ་པ་མང་པོ་སྤྱད་དེ་འཆར་གཞི་ཡོད་པའི་སྒོ་ནས་བོད་སྐད་ཡིག་གི་ལས་བྱེད་མི་སྣ་འདེམ་གསེས་་ཀྱིས་ཕྱིའི་སློབ་གྲྭ་ཆེན་མོ་དང་ཞིབ་འཇུག་སྡེ་ཁག་དུ་ཟབ་སྦྱོང་བྱེད་པར་བཏང་ནས་བྱ་བའི་སྤུས་ཚད་མཐོར་འདེགས་གཏོང་རྒྱུ།

　　དོན་ཚན་ཉེར་བཞི། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་ཆེད་གཉེར་ལོ་ཙཱ་བ་གསོ་སྐྱོང་བྱེད་པར་མཐོང་ཆེན་བྱེད་རྒྱུ།

　　རང་སྐྱོང་ཁུལ་ནང་ཁུལ་ལ་དུ་བོད་ཀྱི་ལས་བཟོ་བ་ཅུང་མང་པའིལས་ཁུངས་དང་བོད་རིགས་འདུས་སྡོད་ཀྱི་ཡུལ་ཚོ་དང༌།གྲོང་བརྡལ་བཅས་ལ་དགོས་མཁོ་དང་བསྟུན་ནས་ཆེད་དུ་ལོ་ཙཱ་བའམ་ཞོར་དུ་འཁུར་ཐུབ་པའི་ལོ་ཙཱ་བ་གང་འོས་བཀོད་སྒྲིག་བྱེད་རྒྱུ།

　　དོན་ཚན་ཉེར་ལྔ། རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་བོད་ཀྱི་སྐད་ཡིག་གི་གཉེར་བའི་མི་སྣར་དུས་ལྟར་བྱ་བ་ཚད་ལྡན་ཡིན་མིན་ལ་བལྟས་ནས་ཐག་བཅད་ལྟར་བྱ་བའི་ཐོབ་ཐང་དང་གོ་ཐོབ་ སོགས་འདེམ་པའི་བྱ་བ་ལེགས་པོར་སྒྲུབ་རྒྱུ།

　　དོན་ཚན་ཉེར་དྲུག རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་སྲོལ་ཡིགའདི་ནང་ལྟར་ལེགས་འགྲུབ་ཡོང་བའི་ལས་ཁུངས་དང་མི་སྒེར་ལ་བྱ་དགའ་སྟེར་བ་དང་དེ་དང་ལྡོག་པ་རྣམས་ལ་ལས་ཁུངས་དང་མི་སྒེར་ལ་སྐྱོན་བརྗོད་སློབ་གསོ་དང་སྲིད་འཛིན་ཐོག་གི་ཆད་པ་གཅོད་རྒྱུ། རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་རྒྱ་བོད་གཉིས་ཡན་ཆད་ཀྱི་ཡིག་རིགས་ཁོང་དུ་ཆུད་པ་བཀོལ་སྤྱོད་གཏོང་པའི་ལས་བྱེད་མི་སྣར་བཟང་བསྟོད་དང༌། བོད་ཀྱི་སྐད་ཡིག་སློབ་ཁྲིད། ཚན་རིག་ཞིབ་འཇུག ཡིག་བསྒྱུར་བཅས་ལ་གྲུབ་འབྲས་ལེགས་པོ་ཐོབ་ཡོད་པའི་མི་སྣར་བྱ་དགའ་སྟེར་རྒྱུ།

　　དོན་ཚན་ཉེར་བདུན། རྨ་ལྷོ་སོག་རིགས་རང་སྐྱོང་རྫོང་གི་སྐད་ཡིག་བྱ་བའི་ཡིག་ཆ་དེ་རང་སྐྱོང་རྫོང་རང་སྐྱོང་ལས་ཁུངས་རང་ཉིད་ཀྱིས་ཐག་གཅོད་དགོས།

　　དོན་ཚན་ཉེར་བརྒྱད། ཡིག་ཆ་འདི་དངོས་སུ་ལག་བསྟར་བྱེད་སྟངས་ཁུལ་མི་དམངས་སྲིད་གཞུང་གིས་ཐག་གཅོད་དགོས།

　　དོན་ཚན་ཉེར་དགུ། ཡིག་ཆ་འདི་རང་སྐྱོང་ཁུལ་མི་དམངས་འཐུས་མི་ཚོགས་ཆེན་གྱི་རྒྱུན་ལས་ཨུ་ཡོནལྷན་ཚོགས་ཀྱིས་འགན་བཞེས་ཏེ་གསལ་འདེབས་བྱེད་དགོས།

　　དོན་ཚན་སུམ་ཅུ། ཡིག་ཆ་འདི་མཚོ་སྔོན་ཞིང་ཆེན་མི་དམངས་འཐུས་མི་ཚོགས་ཆེན་རྒྱུན་ལས་ཨུ་ཡོན་ལྷན་ཚོགས་བརྒྱུད་དེ་མཆོག་མཆན་བཀོད་རྗེས་༡༩༩༤ལོའི་ཟླ་༡པའི་ཚེས་༡ཉིན་ནས་ལག་བསྟར་བྱེད་རྒྱུ།
