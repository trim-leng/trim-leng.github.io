རྩ་ཁྲིམས་ཀྱི་དོན་ཚན་་༡༣༤་ལས་༼མི་རིགས་སོ་སོར་རང་མི་རིགས་ཀྱི་སྐད་ཡིག་སྤྱད་ནས་གཏུག་བཤེར་བྱེད་པའི་དབང་ཆ་ཡོད།མི་དམངས་ཁྲིམས་ཁང་དང་མི་དམངས་ཞིབ་དཔྱོད་ཁང་གིས་ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་མི་ཤེས་པའི་མི་གཏུག་བཤེར་ལ་ཞུགས་ན་དེར་སྐད་སྒྱུར་ཡིག་སྒྱུར་་བྱེད་དགོས།གྲངས་ཉུང་མི་རིགས་འདུས་སྡོད་དང་མི་རིགས་མང་པོ་མཉམ་དུ་འཚོ་སྡོད་ཡོད་པའི་ས་ཆ་རུ།ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་སྤྱད་ནས་གྱོད་གཞི་བརྩད་བཤེར་འཁྲུན་གཅོད་བྱེད་དགོས།ཞུ་གཏུག་དང་ཁྲིམས་ཡིག་་དང་བསྒྲགས་ཡིག་སོགས་དོན་དངོས་ཀྱི་དགོས་མཁོར་དམིགས་ནས་ས་གནས་དེ་ག་འི་ཐུན་མོང་གི་སྐད་ཡིག་རིགས་གཅིག་གམ་ཡིག་རིགས་དུ་མས་འབྲི་དགོས།།༽ཞེས་གཏན་ལ་འཁེལ་ཡོད།

མི་རིགས་མངའ་ཁོངས་རང་སྐྱོང་གི་ཁྲིམས་ཀྱི་དོན་ཚན་་༤༧པའི་ནང་།༼མི་རིགས་མངའ་ཁོངས་རང་སྐྱོང་ས་ཁུལ་གྱི་མིདམངས་ཁྲིམས་ཁང་དང་མི་དམངས་ཞིབ་དཔྱོད་ཁང་གིས་ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཡིག་སྤྱད་ནས་གྱོད་བརྩད་གཅོད་དང་ཁྲིམས་བཤེར་་བྱེད་དགོས་ཤིང་།ས་གནས་དེའི་ཐུན་མོང་སྐད་ཡིག་མི་ཤེས་པ་གཏུག་བཤེར་གང་ཟག་ལ་སྐད་ཡིག་གི་སྒྱུར་གྲོགས་བཙལ་དགོས།།ཁྲིམས་ཡིག་སོགས་ས་གནས་དེ་གའི་སྐད་ཡིག་རིགས་གཅིག་གམ་དུ་མས་འབྲི་དགོསས།མི་རིགས་་ཚང་མར་རང་གི་སྐད་་ཡིག་སྤྱད་ནས་ཞུ་གཏུག་གི་ཐོབ་ཐངཁག་ཐེག་བྱེད་དགོས།༽

མི་དམངས་ཁྲིམས་ཁང་རྩ་འཛུགས་ཀྱི་ཁྲིམས་ཀྱི་དོན་ཚན་༦པར་༼མི་རིགས་སོ་སོར་རང་མི་རིགས་ཀྱི་སྐད་ཡིག་སྤྱད་ནས་གཏུག་བཤེར་བྱེད་པའི་ཐོབ་ཐང་ཡོད།མི་དམངས་ཁྲིམས་ཁང་གིས་ས་གནས་དེ་གའི་ཐུན་མོང་སྐད་ཡིག་མི་ཤེས་པའི་གང་ཟག་གཏུག་བཤེར་ལ་ཞུགས་ན་དེར་སྐད་སྒྱུར་ཡིག་སྒྱུར་་བྱེད་དགོས།གྲངས་ཉུང་མི་རིགས་འདུས་སྡོད་དང་མི་རིགས་མང་པོ་མཉམ་དུ་འཚོ་སྡོད་ཡོད་པའི་ས་ཆ་རུ།ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་སྤྱད་ནས་གྱོད་གཞི་བརྩད་བཤེར་འཁྲུན་གཅོད་བྱེད་དགོས།ཞུ་གཏུག་དང་ཁྲིམས་ཡིག་་དང་བསྒྲགས་ཡིག་སོགས་དོན་དངོས་ཀྱི་དགོས་མཁོར་དམིགས་ནས་ས་གནས་དེ་ག་འི་ཐུན་མོང་གི་སྐད་ཡིག་རིགས་གཅིག་གམ་ཡིག་རིགས་དུ་མས་འབྲི་དགོས།༽ཞེས་འཁོད་ཡོད།

ཉེས་དོན་ཞུ་གཏུག་་ཁྲིམས་ཀྱི་དོན་ཚན་༩པར་༼མི་རིགས་སོ་སོར་རང་མི་རིགས་ཀྱི་སྐད་ཡིག་སྤྱད་ནས་གཏུག་བཤེར་བྱེད་པའི་དབང་ཆ་ཡོད།མི་དམངས་ཁྲིམས་ཁང་དང་མི་དམངས་ཞིབ་དཔྱོད་ཁང་གིས་ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་མི་ཤེས་པའི་མི་གཏུག་བཤེར་ལ་ཞུགས་ན་དེར་སྐད་སྒྱུར་ཡིག་སྒྱུར་་བྱེད་དགོས།གྲངས་ཉུང་མི་རིགས་འདུས་སྡོད་དང་མི་རིགས་མང་པོ་མཉམ་དུ་འཚོ་སྡོད་ཡོད་པའི་ས་ཆ་རུ།ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་སྤྱད་ནས་གྱོད་གཞི་བརྩད་བཤེར་འཁྲུན་གཅོད་བྱེད་དགོས།ཞུ་གཏུག་དང་ཁྲིམས་ཡིག་་དང་བསྒྲགས་ཡིག་སོགས་དོན་དངོས་ཀྱི་དགོས་མཁོར་དམིགས་ནས་ས་གནས་དེ་ག་འི་ཐུན་མོང་གི་སྐད་ཡིག་རིགས་གཅིག་གམ་ཡིག་རིགས་དུ་མས་འབྲི་དགོས།།༽སྐད།　

དམངས་དོན་གཏུར་བཤེར་གྱི་ཁྲིམས་ཀྱི་དོན་ཚན་་༡༡ནང་༼མི་རིགས་སོ་སོར་རང་མི་རིགས་ཀྱི་སྐད་ཡིག་སྤྱད་ནས་གཏུག་བཤེར་བྱེད་པའི་དབང་ཆ་ཡོད།མི་དམངས་ཁྲིམས་ཁང་གིས་ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་མི་ཤེས་པའི་མི་གཏུག་བཤེར་ལ་ཞུགས་ན་དེར་སྐད་སྒྱུར་ཡིག་སྒྱུར་བྱེད་དགོས།གྲངས་ཉུང་མི་རིགས་འདུས་སྡོད་དང་མི་རིགས་མང་པོ་མཉམ་དུ་འཚོ་སྡོད་ཡོད་པའི་ས་ཆ་རུ།ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་སྤྱད་ནས་གྱོད་གཞི་བརྩད་བཤེར་འཁྲུན་གཅོད་བྱེད་དགོས།ཞུ་གཏུག་དང་ཁྲིམས་ཡིག་་དང་བསྒྲགས་ཡིག་སོགས་དོན་དངོས་ཀྱི་དགོས་མཁོར་དམིགས་ནས་ས་གནས་དེ་ག་འི་ཐུན་མོང་གི་སྐད་ཡིག་རིགས་གཅིག་གམ་ཡིག་རིགས་དུ་མས་འབྲི་དགོས།།༽ཟེར།

འཛིན་སྐྱོང་གཏུག་བཤེར་ཀྱི་ཁྲིམས་ཀྱི་དོན་ཚན་་༨པར་༼མི་རིགས་སོ་སོར་རང་མི་རིགས་ཀྱི་སྐད་ཡིག་སྤྱད་ནས་གཏུག་བཤེར་བྱེད་པའི་དབང་ཆ་ཡོད།མི་དམངས་ཁྲིམས་ཁང་གིས་ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་མི་ཤེས་པའི་མི་གཏུག་བཤེར་ལ་ཞུགས་ན་དེར་སྐད་སྒྱུར་ཡིག་སྒྱུར་་བྱེད་དགོས།གྲངས་ཉུང་མི་རིགས་འདུས་སྡོད་དང་མི་རིགས་མང་པོ་མཉམ་དུ་འཚོ་སྡོད་ཡོད་པའི་ས་ཆ་རུ།ས་གནས་དེ་གའི་སྤྱི་སྤྱོད་ཀྱི་སྐད་ཆ་སྤྱད་ནས་གྱོད་གཞི་བརྩད་བཤེར་འཁྲུན་གཅོད་བྱེད་དགོས།ཞུ་གཏུག་དང་ཁྲིམས་ཡིག་་དང་བསྒྲགས་ཡིག་སོགས་དོན་དངོས་ཀྱི་དགོས་མཁོར་དམིགས་ནས་ས་གནས་དེ་ག་འི་ཐུན་མོང་གི་སྐད་ཡིག་རིགས་གཅིག་གམ་ཡིག་རིགས་དུ་མས་འབྲི་དགོས།།༽
